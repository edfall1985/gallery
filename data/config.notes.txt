# CONFIG EXPLANATION

assetPathPrefix = path relatif menuju folder assets dari posisi file HTML
  - Contoh: "../" jika file ada di dalam folder /data
  - Contoh: "" jika file HTML ada di root folder

# OPSIONAL FIELD (untuk dikembangkan selanjutnya)

dataSource = "api" | "local"
  - Menentukan sumber data (backend API atau file lokal di repo)

apiEndpoint = URL endpoint JSON jika menggunakan mode "api"
  - Contoh: "https://datagallery.up.railway.app/dataGallery"

theme = "light" | "dark"
  - Tema UI default saat pertama kali dibuka

requireLogin = true | false
  - Apakah halaman admin perlu login sebelum akses
